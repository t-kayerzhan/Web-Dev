// Bad test example - all tests in one it block
describe("Raises x to power n", function() {
  it("Raises x to the power n", function() {
    let x = 5;
    
    let result = x;
    assert.equal(pow(x, 1), result);
    
    result *= x;
    assert.equal(pow(x, 2), result);
    
    result *= x;
    assert.equal(pow(x, 3), result);
  });
});
